Praktikum

Praktik 1: Bind Mount

Terminal
# Buat folder dan file di host
mkdir -p ~/docker-data
echo "Halo dari host!" > ~/docker-data/pesan.txt

# Jalankan container dengan bind mount
docker run -it -v ~/docker-data:/data ubuntu bash

# Di dalam container, cek file
cat /data/pesan.txt
# Output: Halo dari host!

# Buat file baru dari dalam container
echo "Halo dari container!" > /data/dari-container.txt

# Keluar
exit

# Cek di host - file ada!
cat ~/docker-data/dari-container.txt

Hasil

File yang dibuat di container tersimpan di host! âœ…

Praktik 2: Named Volume

Terminal
# Buat named volume
docker volume create data-saya

# Lihat daftar volume
docker volume ls

# Jalankan container dengan volume
docker run -it -v data-saya:/app/data ubuntu bash

# Di dalam container
echo "Data penting" > /app/data/penting.txt
exit

# Jalankan container BARU dengan volume yang sama
docker run -it -v data-saya:/app/data ubuntu bash

# Data masih ada!
cat /app/data/penting.txt

Terminal
# Jalankan MySQL

  -p 3306:3306 \
